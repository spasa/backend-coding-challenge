Table user
* id integer
  gender smallint
  first_name varchar( 100 )
  last_name varchar( 100 )
* username varchar( 100 )
* password varchar( 200 )
* created_on timestamp
Indexes
Pkpk_user ON id


Table session
* id bigint DEFO nextval('session_seq'::regclass)
* token varchar( 200 )
* user_id integer
* token_type smallint
* create_time timestamp
  access_time timestamp
  expiry_time timestamp
  ip_address varchar( 100 )
* active bool
Indexes
Pkpk_session ON id
 idx_session ON user_id
Uudx_session_0 ON token partial index - where active
Uudx_session_1 ON user_id, token_type partial index - where active
Upk_session_0 ON id, user_id
Foreign Keys
 fk_session_user ( user_id ) ref user (id)


Table expense
* id bigint DEFO nextval('expense_seq'::regclass)
* user_id integer
* session_id bigint
* amount numeric( 19, 4 )
  tax_value numeric( 19, 4 )
* created_on timestamp
Indexes
Pkpk_expense ON id
 idx_expense ON session_id, user_id
 idx_expense_0 ON user_id
Foreign Keys
 fk_expense_session ( session_id, user_id ) ref session (id, user_id)
 fk_expense_user ( user_id ) ref user (id)


Table configuration
* id smallint
* auth_token_duration smallint
* tax_percentage numeric( 19, 2 )
Indexes
Pkpk_configuration ON id
